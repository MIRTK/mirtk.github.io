36 /// Ordered list of parameter name/value pairs
37 typedef Array<Pair<string, string> > ParameterList;

41 /**
42  * Base class for all MIRTK object classes
43  *
44  * \note This base class must be a virtual interface without any own data
45  * members to not change the type size of subclasses! Derive another
46  * intermediate abstract base class from it to add data members shared
47  * by a more specific class of objects.
48  */

51 public:

68  /// Set parameters from name/value pairs
69  inline void Parameter(const ParameterList &param) {
70  for (ParameterConstIterator it = param.begin(); it != param.end(); ++it) {
71  this->Set(it->first.c_str(), it->second.c_str());
72  }
73  }

115 /// Insert/replace value into/in parameters list
116 template <class T>
117 inline ParameterList &Insert(ParameterList &params, string name, T value)
118 {
119  ParameterIterator pos = Find(params, name);
120  if (pos == params.end()) {
121  params.push_back(make_pair(name, ToString(value)));
122  } else {
123  pos->second = ToString(value);
124  }
125  return params;
126 }

156 // -----------------------------------------------------------------------------
157 /// Insert/replace values into/in parameters list
158 inline ParameterList &Insert(ParameterList &params,
159  const ParameterList &other,
160  const char *prefix = NULL)
161 {
162  if (prefix) {
163  string name;
164  for (ParameterConstIterator it = other.begin(); it != other.end(); ++it) {
165  name = it->first;
166  name[0] = ::tolower(name[0]);
167  Insert(params, string(prefix) + " " + name, it->second);
168  }
169  } else {
170  for (ParameterConstIterator it = other.begin(); it != other.end(); ++it) {
171  Insert(params, it->first, it->second);
172  }
173  }
174  return params;
175 }

177 // -----------------------------------------------------------------------------
178 /// Remove parameter from parameters list
179 inline ParameterList &Remove(ParameterList &params, string name)
180 {
181  ParameterIterator pos = Find(params, name);
182  if (pos != params.end()) params.erase(pos);
183  return params;
184 }

349 /// Define pointer to component (cf. UML composition) which can also be set to
350 /// an externally managed instance instead using the corresponding accessors
351 ///
352 /// A loose component is a pointer that must never by a nullptr. It must be
353 /// initialized during construction of an instance of the class which this
354 /// component attribute belongs.
355 ///
356 /// Unlike a regular component, a loose component can alternatively be set to an
357 /// externally instantiated object of the component type. In this case, the class
358 /// revokes ownership of the object which must be deleted by the client code.
359 /// When a nullptr is given as argument to the setter of the loose component,
360 /// the current pointer is replaced by a new copy that is owned by this class.
361 /// It can thus be used to release the external object again before destruction
362 /// in cases where the class that the loose component belongs to remains in use.
363 #define mirtkLooseComponentMacro(type, name) \
364  protected: \
365  type *_##name; \
366  bool _##name##Owner; \
367  public: \
368  /** Set pointer to _##name attribute */ \
369  inline virtual void name(type *arg) { \
370  if (_##name##Owner) delete _##name; \
371  if (arg) { \
372  _##name = arg; \
373  _##name##Owner = false; \
374  } else { \
375  _##name = new type(*_##name); \
376  _##name##Owner = true; \
377  } \
378  } \
379  /** Get pointer to _##name attribute */ \
380  inline type *name() { return _##name; } \
381  /** Get const pointer to _##name attribute */ \
382  inline const type *name() const { return _##name; } \
