225 inline void LinearFreeFormTransformation3D
226 ::LocalTransform(double &x, double &y, double &z, double, double) const
227 {
228  // Convert to lattice coordinates
229  double dx = x, dy = y, dz = z;
230  this->WorldToLattice(dx, dy, dz);
231  // Evaluate displacement
232  this->Evaluate(dx, dy, dz);
233  // Transform point
234  x += dx, y += dy, z += dz;
235 }

255 // -----------------------------------------------------------------------------
256 inline void LinearFreeFormTransformation3D
257 ::LocalJacobian(Matrix &jac, double x, double y, double z, double, double) const
258 {
259  // Convert to lattice coordinates
260  this->WorldToLattice(x, y, z);
261  // Compute 1st order derivatives
262  this->EvaluateJacobian(jac, x, y, z);
263  // Convert derivatives to world coordinates
264  JacobianToWorld(jac);
265  // Add derivatives of "x" term in T(x) = x + this->Evaluate(x)
266  jac(0, 0) += 1.0;
267  jac(1, 1) += 1.0;
268  jac(2, 2) += 1.0;
269 }
270 
271 // -----------------------------------------------------------------------------
272 inline void LinearFreeFormTransformation3D
273 ::JacobianDOFs(double jac[3], int ci, int cj, int ck, double x, double y, double z) const
274 {
275  // Convert point to lattice coordinates
276  this->WorldToLattice(x, y, z);
277  // Calculate tensor product of 1D linear interpolation kernels
278  const double dx = abs(ci - x);
279  const double dy = abs(cj - y);
280  const double dz = abs(ck - z);
281  if (dx < 1.0 && dy < 1.0 && dz < 1.0) {
282  jac[0] = jac[1] = jac[2] = (1.0 - dx) * (1.0 - dy) * (1.0 - dz);
283  } else {
284  jac[0] = jac[1] = jac[3] = .0;
285  }
286 }
