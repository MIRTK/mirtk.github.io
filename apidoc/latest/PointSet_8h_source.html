403 // -----------------------------------------------------------------------------
404 inline void PointSet::Resize(int n, const Point &p)
405 {
406  int m = _m;
407  while (n > m) m += POINTSET_SIZE;
408  this->Reserve(m);
409  _n = n;
410  for (int i = _n; i < _m; ++i) _data[i] = p;
411 }
412 
413 // -----------------------------------------------------------------------------
414 inline void PointSet::Reserve(int m)
415 {
416  if (_m < m) {
417  _m = m;
418  Point *new_data = Allocate<Point>(m);
419  for (int i = 0; i < _n; ++i) new_data[i] = _data[i];
420  Deallocate(_data);
421  _data = new_data;
422  }
423 }
