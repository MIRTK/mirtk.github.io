386 // -----------------------------------------------------------------------------
387 inline void BSplineFreeFormTransformation3D
388 ::EvaluateDerivativeOfJacobianWrtDOF(Matrix &dJdp, int dof, double x, double y) const
389 {
390  const int dim = this->DOFToDimension(dof);
391 
392  int i, j;
393  this->IndexToLattice(this->DOFToIndex(dof), i, j);
394 
395  double val;
396  if (dim == 0) val = Kernel::B_I(x - i) * Kernel::B (y - j);
397  else val = Kernel::B (x - i) * Kernel::B_I(y - j);
398 
399  dJdp.Initialize(3, 3);
400  dJdp(0, dim) = dJdp(1, dim) = dJdp(2, dim) = val;
401 }
402 
403 // -----------------------------------------------------------------------------
404 inline void BSplineFreeFormTransformation3D
405 ::EvaluateDerivativeOfJacobianWrtDOF(Matrix &dJdp, int dof, double x, double y, double z) const
406 {
407  const int dim = this->DOFToDimension(dof);
408 
409  int i, j, k;
410  this->IndexToLattice(this->DOFToIndex(dof), i, j, k);
411 
412  double val;
413  if (dim == 0) val = Kernel::B_I(x - i) * Kernel::B (y - j) * Kernel::B (z - k);
414  else if (dim == 1) val = Kernel::B (x - i) * Kernel::B_I(y - j) * Kernel::B (z - k);
415  else val = Kernel::B (x - i) * Kernel::B (y - j) * Kernel::B_I(z - k);
416 
417  dJdp.Initialize(3, 3);
418  dJdp(0, dim) = dJdp(1, dim) = dJdp(2, dim) = val;
419 }

425 // -----------------------------------------------------------------------------
426 inline void BSplineFreeFormTransformation3D
427 ::LocalTransform(double &x, double &y, double &z, double, double) const
428 {
429  // Convert to lattice coordinates
430  double dx = x, dy = y, dz = z;
431  this->WorldToLattice(dx, dy, dz);
432  // Evaluate displacement
433  Evaluate(dx, dy, dz);
434  // Transform point
435  x += dx, y += dy, z += dz;
436 }

443 inline void BSplineFreeFormTransformation3D::FFDJacobianWorld(Matrix &jac, double x, double y, double z, double, double) const
444 {
445  // Convert to lattice coordinates
446  this->WorldToLattice(x, y, z);
447  // Compute 1st order derivatives
448  if (_z == 1) EvaluateJacobianWorld(jac, x, y);
449  else EvaluateJacobianWorld(jac, x, y, z);
450  // Add derivatives of "x" term in T(x) = x + FFD(x)
451  jac(0, 0) += 1.0;
452  jac(1, 1) += 1.0;
453  jac(2, 2) += 1.0;
454 }
