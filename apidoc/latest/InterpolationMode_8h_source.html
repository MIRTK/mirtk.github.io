57 template <>
58 inline string ToString(const InterpolationMode &m, int w, char c, bool left)
59 {
60  const char *str;
61  switch(m) {
62  case Interpolation_NN: str = "NN"; break;
63  case Interpolation_Linear: str = "Linear"; break;
64  case Interpolation_FastLinear: str = "Fast linear"; break;
65  case Interpolation_BSpline: str = "BSpline"; break;
66  case Interpolation_CSpline: str = "CSpline"; break;
67  case Interpolation_CubicBSpline: str = "Cubic BSpline"; break;
68  case Interpolation_FastCubicBSpline: str = "Fast cubic BSpline"; break;
69  case Interpolation_SBased: str = "SBased"; break;
70  case Interpolation_Sinc: str = "Sinc"; break;
71  case Interpolation_Gaussian: str = "Gaussian"; break;
72  case Interpolation_NNWithPadding: str = "NN with padding"; break;
73  case Interpolation_LinearWithPadding: str = "Linear with padding"; break;
74  case Interpolation_FastLinearWithPadding: str = "Fast linear with padding"; break;
75  case Interpolation_BSplineWithPadding: str = "BSpline with padding"; break;
76  case Interpolation_CubicBSplineWithPadding: str = "Cubic BSpline with padding"; break;
77  case Interpolation_FastCubicBSplineWithPadding: str = "Fast cubic BSpline with padding"; break;
78  case Interpolation_CSplineWithPadding: str = "CSpline with padding"; break;
79  case Interpolation_SBasedWithPadding: str = "SBased with padding"; break;
80  case Interpolation_SincWithPadding: str = "Sinc with padding"; break;
81  case Interpolation_GaussianWithPadding: str = "Gaussian with padding"; break;
82  default: str = "Unknown"; break;
83  }
84  return ToString(str, w, c, left);
85 }

88 template <>
89 inline bool FromString(const char *str, InterpolationMode &m)
90 {
91  const string lstr = ToLower(str);
92  if (lstr == "nn") m = Interpolation_NN;
93  else if (lstr == "linear") m = Interpolation_Linear;
94  else if (lstr == "fast linear") m = Interpolation_FastLinear;
95  else if (lstr == "bspline") m = Interpolation_BSpline;
96  else if (lstr == "cspline") m = Interpolation_CSpline;
97  else if (lstr == "cubic bspline") m = Interpolation_CubicBSpline;
98  else if (lstr == "fast cubic bspline") m = Interpolation_FastCubicBSpline;
99  else if (lstr == "sbased") m = Interpolation_SBased;
100  else if (lstr == "sinc") m = Interpolation_Sinc;
101  else if (lstr == "gaussian") m = Interpolation_Gaussian;
102  else if (lstr == "nn with padding") m = Interpolation_NNWithPadding;
103  else if (lstr == "nn (with padding)") m = Interpolation_NNWithPadding;
104  else if (lstr == "linear with padding") m = Interpolation_LinearWithPadding;
105  else if (lstr == "linear (with padding)") m = Interpolation_LinearWithPadding;
106  else if (lstr == "fast linear with padding") m = Interpolation_FastLinearWithPadding;
107  else if (lstr == "fast linear (with padding)") m = Interpolation_FastLinearWithPadding;
108  else if (lstr == "bspline with padding") m = Interpolation_BSplineWithPadding;
109  else if (lstr == "bspline (with padding)") m = Interpolation_BSplineWithPadding;
110  else if (lstr == "cubic bspline with padding") m = Interpolation_CubicBSplineWithPadding;
111  else if (lstr == "cubic bspline (with padding)") m = Interpolation_CubicBSplineWithPadding;
112  else if (lstr == "fast cubic bspline with padding") m = Interpolation_FastCubicBSplineWithPadding;
113  else if (lstr == "fast cubic bspline (with padding)") m = Interpolation_FastCubicBSplineWithPadding;
114  else if (lstr == "cspline with padding") m = Interpolation_CSplineWithPadding;
115  else if (lstr == "cspline (with padding)") m = Interpolation_CSplineWithPadding;
116  else if (lstr == "sbased with padding") m = Interpolation_SBasedWithPadding;
117  else if (lstr == "sbased (with padding)") m = Interpolation_SBasedWithPadding;
118  else if (lstr == "sinc with padding") m = Interpolation_SincWithPadding;
119  else if (lstr == "sinc (with padding)") m = Interpolation_SincWithPadding;
120  else if (lstr == "gaussian with padding") m = Interpolation_GaussianWithPadding;
121  else if (lstr == "gaussian (with padding)") m = Interpolation_GaussianWithPadding;
122  else return false;
123  return true;
124 }
