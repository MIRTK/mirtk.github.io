139 // -----------------------------------------------------------------------------
140 inline bool FundamentalMap::AddSourcePoint(double p[3], double tol)
141 {
142  if (tol > .0) {
143  for (int i = 0; i < _SourcePoints.Size(); ++i) {
144  const Point &q = _SourcePoints(i);
145  if (fequal(p[0], q._x, tol) &&
146  fequal(p[1], q._y, tol) &&
147  fequal(p[2], q._z, tol)) {
148  return false;
149  }
150  }
151  }
152  _SourcePoints.Add(p);
153  _Coefficients.Resize(_SourcePoints.Size(), _Coefficients.Cols());
154  return true;
155 }
