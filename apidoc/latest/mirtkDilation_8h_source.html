64 template <class TVoxel>
65 void Dilate(BaseImage *image, int iterations, ConnectivityType connectivity)
66 {
67  GenericImage<TVoxel> * const im = dynamic_cast<GenericImage<TVoxel> *>(image);
68  mirtkAssert(im != nullptr, "template function called with correct type");
69  Dilation<TVoxel> dilation;
70  dilation.Connectivity(connectivity);
71  dilation.Input (im);
72  dilation.Output(im);
73  for (int i = 0; i < iterations; ++i) dilation.Run();
74 }
