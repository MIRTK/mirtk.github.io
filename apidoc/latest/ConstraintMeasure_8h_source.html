30 /// Enumeration of available transformation regularization terms
31 ///
32 /// \note This enumeration constains only a subset of all EnergyMeasure
33 /// enumeration values, whereby the integer value of corresponding
34 /// enumeration values is equal.
35 ///
36 /// \sa EnergyMeasure
37 enum ConstraintMeasure
38 {
39  CM_Unknown = CM_Begin, ///< Unknown/invalid regularizer
40  CM_VolumePreservation = EM_VolumePreservation, ///< Volume preservation constraint
41  CM_TopologyPreservation = EM_TopologyPreservation, ///< Topology preservation constraint
42  CM_Sparsity = EM_Sparsity, ///< Default sparsity constraint
43  CM_BendingEnergy = EM_BendingEnergy, ///< Thin-plate spline bending energy
44  CM_L0Norm = EM_L0Norm, ///< Sparsity constraint based on l0-norm
45  CM_L1Norm = EM_L1Norm, ///< Sparsity constraint based on l1-norm
46  CM_L2Norm = EM_L2Norm, ///< Sparsity constraint based on l2-norm
47  CM_SqLogDetJac = EM_SqLogDetJac, ///< Squared logarithm of the Jacobian determinant
48  CM_MinDetJac = EM_MinDetJac, ///< Constrain minimum Jacobian determinant
49 };
50 
51 // -----------------------------------------------------------------------------
52 template <>
53 inline string ToString(const ConstraintMeasure &cm, int w, char c, bool left)
54 {
55  EnergyMeasure em = static_cast<EnergyMeasure>(cm);
56  if (em <= CM_Begin || em >= CM_End) return ToString("Unknown", w, c, left);
57  return ToString(em, w, c, left);
58 }
59 
60 // -----------------------------------------------------------------------------
61 template <>
62 inline bool FromString(const char *str, ConstraintMeasure &cm)
63 {
64  EnergyMeasure em = EM_Unknown;
65  if (FromString(str, em) && CM_Begin < em && em < CM_End) {
66  cm = static_cast<ConstraintMeasure>(em);
67  return true;
68  } else {
69  cm = CM_Unknown;
70  return false;
71  }
72 }
