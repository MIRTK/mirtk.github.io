32 /**
33  * Class for multi-level FFD where global and local transformations are composed.
34  *
35  * T_mffd(x) = T_affine ° T_local^N ° T_local^N-1 ° ... ° T_local^1 ° T_global(x)
36  *
37  * where T_global is the initial global affine transformation, T_local are
38  * the non-rigid free-form deformations (FFD), and T_affine is an optional
39  * affine transformation which is applied after the deformations.
40  */

255  /// Applies the chain rule to convert spatial non-parametric gradient
256  /// to a gradient w.r.t the parameters of this transformation.
257  ///
258  /// If the transformation itself is non-parametric, the gradient will be passed through
259  /// unchanged. The default implementation uses the full Jacobian matrix computed for each
260  /// DoF separately (i.e., calls JacobianDOFs for each DoF).
261  ///
262  /// For 4D transformations, the temporal coordinate t used for the computation of the Jacobian
263  /// of the transformation w.r.t the transformation parameters for all spatial (x, y, z) voxel
264  /// coordinates in world units, is assumed to correspond to the temporal origin of the given
265  /// gradient image. For 4D transformations parameterized by velocities, a second time for the
266  /// upper integration bound can be provided as last argument to this method. This last
267  /// argument is ignored by transformations parameterized by displacements.
268  ///
269  /// \sa ImageSimilarityMetric::EvaluateGradient
270  virtual void ParametricGradient(const GenericImage<double> *, double *,
271  const WorldCoordsImage * = NULL,
272  const WorldCoordsImage * = NULL,
273  double = -1, double = 1) const;

336 inline bool FluidFreeFormTransformation
337 ::DOFBoundingBox(const Image *image, int dof, int &i1, int &j1, int &k1,
338  int &i2, int &j2, int &k2, double fraction) const
339 {
340  // Get local transformation and corresponding parameter index
341  const FreeFormTransformation *ffd;
342  const int n = DOFIndexToLocalTransformation(this, dof, ffd, dof);
343 
344  // Calculate bounding box in world coordinates
345  double x1, y1, z1, x2, y2, z2;
346  ffd->BoundingBox(ffd->DOFToIndex(dof), x1, y1, z1, x2, y2, z2, fraction);
347 
348  // Apply inverse transformations which preceed this local transformation
349  this->Inverse(-1, n, x1, y1, z1);
350  this->Inverse(-1, n, x2, y2, z2);
351 
352  // Convert to image coordinates
353  image->WorldToImage(x1, y1, z1);
354  image->WorldToImage(x2, y2, z2);
355 
356  if (x2 < x1) swap(x1, x2);
357  if (y2 < y1) swap(y1, y2);
358  if (z2 < z1) swap(z1, z2);
359 
360  // Round up/down to nearest voxel in image domain
361  i1 = static_cast<int>(ceil (x1));
362  i2 = static_cast<int>(floor(x2));
363  j1 = static_cast<int>(ceil (y1));
364  j2 = static_cast<int>(floor(y2));
365  k1 = static_cast<int>(ceil (z1));
366  k2 = static_cast<int>(floor(z2));
367  // When both indices are outside in opposite directions,
368  // use the full range [0, N[. If they are both outside in
369  // the same direction, the condition i1 <= i2 is false which
370  // indicates that the bounding box is empty in this case
371  i1 = (i1 < 0 ? 0 : (i1 >= image->X() ? image->X() : i1));
372  i2 = (i2 < 0 ? -1 : (i2 >= image->X() ? image->X() - 1 : i2));
373  j1 = (j1 < 0 ? 0 : (j1 >= image->Y() ? image->Y() : j1));
374  j2 = (j2 < 0 ? -1 : (j2 >= image->Y() ? image->Y() - 1 : j2));
375  k1 = (k1 < 0 ? 0 : (k1 >= image->Z() ? image->Z() : k1));
376  k2 = (k2 < 0 ? -1 : (k2 >= image->Z() ? image->Z() - 1 : k2));
377  return i1 <= i2 && j1 <= j2 && k1 <= k2;
378 }
