71 #if MIRTK_Image_WITH_VTK
72  /// Process given vtkDataArray
73  virtual void Process(vtkDataArray *data, bool *mask = NULL)
74  {
75  const int n = static_cast<int>(data->GetNumberOfTuples() * data->GetNumberOfComponents());
76  if (data->GetDataType() == VTK_DOUBLE) {
77  this->Process(n, reinterpret_cast<double *>(data->GetVoidPointer(0)), mask);
78  } else {
79  double *d = new double[n];
80  double *tuple = d;
81  for (vtkIdType i = 0; i < data->GetNumberOfTuples(); ++i) {
82  data->GetTuple(i, tuple);
83  tuple += data->GetNumberOfComponents();
84  }
85  this->Process(n, d, mask);
86  for (vtkIdType i = 0; i < data->GetNumberOfTuples(); ++i) {
87  data->SetTuple(i, tuple);
88  tuple += data->GetNumberOfComponents();
89  }
90  delete[] d;
91  }
92  }
93 #endif
94 };

184 // -----------------------------------------------------------------------------
185 // Add Calculate function that takes a vtkDataArray as argument
186 // and computes a single return value
187 //
188 // Subclass must implement:
189 // template <class T> double Calculate(int n, const T *data, const bool *mask)
190 #if MIRTK_Image_WITH_VTK
191  #define mirtkCalculateVtkDataArray1() \
192  static double Calculate(vtkDataArray *data, const bool *mask = NULL) \
193  { \
194  const void *p = data->GetVoidPointer(0); \
195  const int n = static_cast<int>(data->GetNumberOfTuples() * \
196  data->GetNumberOfComponents()); \
197  switch (data->GetDataType()) { \
198  case VTK_SHORT: return Calculate(n, reinterpret_cast<const short *>(p), mask); \
199  case VTK_INT: return Calculate(n, reinterpret_cast<const int *>(p), mask); \
200  case VTK_FLOAT: return Calculate(n, reinterpret_cast<const float *>(p), mask); \
201  case VTK_DOUBLE: return Calculate(n, reinterpret_cast<const double *>(p), mask); \
202  default: \
203  cerr << "Unsupported vtkDataArray type: " << data->GetDataType() << endl; \
204  exit(1); \
205  } \
206  return numeric_limits<double>::quiet_NaN(); \
207  }
208 #else
209  #define mirtkCalculateVtkDataArray1()
210 #endif
