39 /**
40  * Auxiliary class used to find self-collisions of a triangulated surface mesh
41  *
42  * Instances of this class detect different types of self-collisions such as
43  * in particular self-intersections between non-adjacent as well as adjacent
44  * triangular faces and a list of non-adjacent faces which are about to collide,
45  * i.e., very close to each other. They are used to impose either hard or soft
46  * non-self-intersection constraints on a deformable surface.
47  */
48 class SurfaceCollisions : public Object
49 {
50  mirtkObjectMacro(SurfaceCollisions);

56  /// Type of self-collision
57  enum CollisionType
58  {
59  NoCollision, ///< No collision with another triangle within search range
60  Collision, ///< Unknown or mixed collision types
61  FrontfaceCollision, ///< Second triangle is within minimum distance towards the front of first triangle
62  BackfaceCollision, ///< Second triangle is within minimum distance towards the back of first triangle
63  Intersection, ///< Unknown or mixed intersection types
64  SelfIntersection, ///< Non-adjacent triangles intersect each other
65  AdjacentIntersection, ///< Adjacent triangles intersect each other
66  Ambiguous ///< Both collisions and self-intersections found
67  };

75  /// Structure storing information about detected self-intersection
76  struct IntersectionInfo
77  {
78  int _CellId; ///< ID of other triangle
79  bool _Adjacent; ///< Whether triangles are adjacent
80 
81  IntersectionInfo(int cellId = -1, bool adj = false)
82  :
83  _CellId(cellId), _Adjacent(adj)
84  {}
85 
86  IntersectionInfo(const IntersectionInfo &other)
87  :
88  _CellId(other._CellId), _Adjacent(other._Adjacent)
89  {}
90 
91  IntersectionInfo &operator =(const IntersectionInfo &other)
92  {
93  _CellId = other._CellId;
94  _Adjacent = other._Adjacent;
95  return *this;
96  }
97 
98  bool operator <(const IntersectionInfo &rhs) const
99  {
100  return _CellId < rhs._CellId;
101  }
102  };

104  /// Structure storing information about detected collision
105  struct CollisionInfo
106  {
107  int _CellId; ///< ID of other triangle
108  double _Point1[3]; ///< Point in this triangle which is closest
109  double _Point2[3]; ///< Point in other triangle which is closest
110  double _Distance; ///< Distance between closest points
111  CollisionType _Type; ///< Type of collision
112 
113  CollisionInfo(int cellId = -1)
114  :
115  _CellId(cellId),
116  _Distance(numeric_limits<double>::quiet_NaN()),
117  _Type(Collision)
118  {
119  memset(_Point1, 0, 3 * sizeof(double));
120  memset(_Point2, 0, 3 * sizeof(double));
121  }
122 
123  CollisionInfo(int cellId, const double p1[3], const double p2[3], CollisionType type = Collision)
124  :
125  _CellId(cellId),
126  _Type(type)
127  {
128  memcpy(_Point1, p1, 3 * sizeof(double));
129  memcpy(_Point2, p2, 3 * sizeof(double));
130  const double a = _Point2[0] - _Point1[0];
131  const double b = _Point2[1] - _Point1[1];
132  const double c = _Point2[2] - _Point1[2];
133  _Distance = sqrt(a * a + b * b + c * c);
134  }
135 
136  CollisionInfo(int cellId, const double p1[3], const double p2[3], double dist, CollisionType type = Collision)
137  :
138  _CellId(cellId),
139  _Distance(dist),
140  _Type(type)
141  {
142  memcpy(_Point1, p1, 3 * sizeof(double));
143  memcpy(_Point2, p2, 3 * sizeof(double));
144  }
145 
146  CollisionInfo(const CollisionInfo &other)
147  :
148  _CellId(other._CellId),
149  _Distance(other._Distance),
150  _Type(other._Type)
151  {
152  memcpy(_Point1, other._Point1, 3 * sizeof(double));
153  memcpy(_Point2, other._Point2, 3 * sizeof(double));
154  }
155 
156  CollisionInfo &operator =(const CollisionInfo &other)
157  {
158  _CellId = other._CellId;
159  _Distance = other._Distance;
160  _Type = other._Type;
161  memcpy(_Point1, other._Point1, 3 * sizeof(double));
162  memcpy(_Point2, other._Point2, 3 * sizeof(double));
163  return *this;
164  }
165 
166  bool operator <(const CollisionInfo &rhs) const
167  {
168  return _CellId < rhs._CellId;
169  }
170  };

347 // -----------------------------------------------------------------------------
348 inline bool SurfaceCollisions::IsCollision(CollisionType type)
349 {
350  return (Collision <= type && type < Intersection) || (type == Ambiguous);
351 }
352 
353 // -----------------------------------------------------------------------------
354 inline bool SurfaceCollisions::IsIntersection(CollisionType type)
355 {
356  return Intersection <= type && type <= Ambiguous;
357 }
