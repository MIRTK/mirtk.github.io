10 ##############################################################################
11 # @file MIRTKUse.cmake
12 # @brief CMake package configuration use file.
13 #
14 # @note The MIRTKUse.cmake file is automatically generated
15 # by BASIS from the template file ConfigUse.cmake.in which is part of the
16 # BASIS installation.
17 #
18 # @ingroup BasisConfig
19 ##############################################################################
20 
21 # ============================================================================
22 # namespace
23 # ============================================================================
24 
25 # prefix used for variable names
26 set (NS "MIRTK_")
27 
28 # allow caller to change namespace - used by projects with modules
29 if (${NS}CONFIG_PREFIX)
30  set (NS "${${NS}CONFIG_PREFIX}")
31 endif ()
32 
33 # ============================================================================
34 # include guard
35 # ============================================================================
36 
37 if (__${NS}USE_FILE_INCLUDED)
38  return ()
39 else ()
40  set (__${NS}USE_FILE_INCLUDED 1)
41 endif ()
42 
43 # ============================================================================
44 # use project settings
45 # ============================================================================
46 
47 # include directories
48 if (${NS}INCLUDE_DIRS)
49  include_directories (${${NS}INCLUDE_DIRS})
50 elseif (${NS}INCLUDE_DIR)
51  include_directories (${${NS}INCLUDE_DIR})
52 endif ()
53 
54 # library directories
55 if (${NS}LIBRARY_DIRS)
56  link_directories (${${NS}LIBRARY_DIRS})
57 elseif (${NS}LIBRARY_DIR)
58  link_directories (${${NS}LIBRARY_DIR})
59 endif ()
60 
61 # ============================================================================
62 # import build targets
63 # ============================================================================
64 
65 ## @brief Whether to import the exported targets.
66 set (NO_${NS}IMPORTS "${NO_${NS}IMPORTS}")
67 
68 if (NOT NO_${NS}IMPORTS)
69  include ("${${NS}EXPORTS_FILE}" OPTIONAL)
70  include ("${${NS}CUSTOM_EXPORTS_FILE}" OPTIONAL)
71 endif ()
72 
73 # ============================================================================
74 # use modules
75 # ============================================================================
76 
77 foreach (M IN LISTS ${NS}MODULES_REQUESTED)
78  if (${NS}${M}_USE_FILE)
79  set (${M}_CONFIG_PREFIX "${NS}${M}_")
80  include ("${${NS}${M}_USE_FILE}")
81  unset (${M}_CONFIG_PREFIX)
82  endif ()
83 endforeach ()
