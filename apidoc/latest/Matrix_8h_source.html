729 inline bool Matrix::operator ==(const Matrix &m) const
730 {
731  if ((m._rows != _rows) || (m._cols != _cols)) return false;
732  const int n = this->NumberOfElements();
733  const double *ptr1 = m . RawPointer();
734  const double *ptr2 = this->RawPointer();
735  for (int i = 0; i < n; ++i, ++ptr1, ++ptr2) {
736  if (!fequal(*ptr2, *ptr1)) return false;
737  }
738  return true;
739 }

841 // -----------------------------------------------------------------------------
842 inline void Matrix::RowRange(int r, double &v1, double &v2) const
843 {
844  if (_rows > 0) {
845  v1 = v2 = _matrix[0][r];
846  for (int c = 1; c < _cols; ++c) {
847  const double &v = _matrix[c][r];
848  v1 = min(v1, v), v2 = max(v2, v);
849  }
850  } else {
851  v1 = v2 = .0;
852  }
853 }

920 {
921  int i, j;
922  double normInf = -1.0 * DBL_MAX;
923  double sum;
924

926  sum = 0;
927  for (j = 0; j < _cols; ++j) {
928  sum += abs(_matrix[j][i]);
929  }

932  }
