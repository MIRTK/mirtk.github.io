257  _maskOwner = false;
258  }
259  if (image.HasBackgroundValue()) {

374 template <class VoxelType> template <class VoxelType2>
375 bool HashImage<VoxelType>::operator==(const HashImage<VoxelType2> &image) const
376 {
377  if (this->GetImageAttributes() != image.GetImageAttributes()) return false;
378  if (_DefaultValue != image.DefaultValue()) return false;
379  if (_Data.size() != image.NumberOfNonDefaultVoxels()) return false;
380 
381  int idx;
382  for (int idx = 0; idx < _NumberOfVoxels; ++idx) {
383  if (IsForeground(idx) && image.IsForeground(idx)){
384  if(Get(idx)!= voxel_cast<VoxelType>(image.Get(idx)))
385  return false;
386  }
387  }
388  return true;
389 }

395 // -----------------------------------------------------------------------------
396 template <class VoxelType>
397 void HashImage<VoxelType>
398 ::GetRegion(HashImage &image, int k, int m) const
399 {
400  int i, j;
401  double x1, y1, z1, t1, x2, y2, z2, t2;
402 
403  if ((k < 0) || (k >= _attr._z) || (m < 0) || (m >= _attr._t)) {
404  cerr << "HashImage::GetRegion: Parameter out of range" << endl;
405  exit(1);
406  }
407 
408  // Initialize
409  ImageAttributes attr = this->Attributes();
410  attr._z = 1;
411  attr._t = 1;
412  attr._xorigin = 0;
413  attr._yorigin = 0;
414  attr._zorigin = 0;
415  image.Initialize(attr);
416 
417  // Calculate position of first voxel in roi in original image
418  x1 = 0;
419  y1 = 0;
420  z1 = k;
421  this->ImageToWorld(x1, y1, z1);
422  t1 = this->ImageToTime(m);
423 
424  // Calculate position of first voxel in roi in new image
425  x2 = 0;
426  y2 = 0;
427  z2 = 0;
428  t2 = 0;
429  image.ImageToWorld(x2, y2, z2);
430  t2 = image.ImageToTime(0);
431 
432  // Shift origin of new image accordingly
433  image.PutOrigin(x1 - x2, y1 - y2, z1 - z2, t1 - t2);
434 
435  // Copy region
436  VoxelType value, empty=image.DefaultValue();
437  for (j = 0; j < _attr._y; j++)
438  for (i = 0; i < _attr._x; i++) {
439  value=Get(i,j,k,m);
440  if(value!=empty) image.Put(i,j,0,0, value );
441  }
442 }

471 ::GetRegion(HashImage &image, int i1, int j1, int k1,
472  int i2, int j2, int k2) const

475  double x1, y1, z1, x2, y2, z2;
476 
477  if ((i1 < 0) || (i1 >= i2) ||

480  (i2 > _attr._x) || (j2 > _attr._y) || (k2 > _attr._z)) {
481  cerr << "HashImage::GetRegion: Parameter out of range\n";
482  exit(1);
483  }
484 
485  // Initialize
486  ImageAttributes attr = this->Attributes();
487  attr._x = i2 - i1;
488  attr._y = j2 - j1;
489  attr._z = k2 - k1;
490  attr._xorigin = 0;
491  attr._yorigin = 0;
492  attr._zorigin = 0;
493  image.Initialize(attr);
494 
495  // Calculate position of first voxel in roi in original image
496  x1 = i1;
497  y1 = j1;
498  z1 = k1;
499  this->ImageToWorld(x1, y1, z1);
500 
501  // Calculate position of first voxel in roi in new image
502  x2 = 0;
503  y2 = 0;
504  z2 = 0;
505  image.ImageToWorld(x2, y2, z2);
506 
507  // Shift origin of new image accordingly
508  image.PutOrigin(x1 - x2, y1 - y2, z1 - z2);
509 
510  // Copy region
511  VoxelType value, empty=image.DefaultValue();
512  for (l = 0; l < _attr._t; l++)
513  for (k = k1; k < k2; k++)
514  for (j = j1; j < j2; j++)
515  for (i = i1; i < i2; i++) {

517  if(value!=empty) image.Put(i-i1,j-j1,k-k1,l, value );

627 {

648  for (int j = 0; j < _attr._y; j++)
