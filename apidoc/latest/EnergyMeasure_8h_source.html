245 /// Convert energy measure string to enumeration value
246 template <>
247 inline bool FromString(const char *str, EnergyMeasure &value)
248 {
249  const string lstr = ToLower(Trim(str));
250 
251  value = EM_Unknown;
252 
253  // ---------------------------------------------------------------------------
254  // Alternative names for image (dis-)similarity measures
255  if (value == EM_Unknown) {
256  if (lstr == "ncc") value = EM_LNCC;
257  else if (lstr == "lcc") value = EM_LNCC;
258  else if (lstr == "kappa") value = EM_K;
259  else if (lstr == "correlation ratio xy" ||
260  lstr == "correlationratioxy") value = EM_CR_XY;
261  else if (lstr == "correlation ratio" ||
262  lstr == "correlationratio" ||
263  lstr == "correlation ratio yx" ||
264  lstr == "correlationratioyx") value = EM_CR_YX;
265  }
266 
267  // ---------------------------------------------------------------------------
268  // Alternative names for point set distance measures
269  if (value == EM_Unknown) {
270  if (lstr == "fiducial error" || lstr == "fiducialerror" ||
271  lstr == "landmark error" || lstr == "landmarkerror" ||
272  lstr == "landmark registration error" || lstr == "landmarkregistrationerror") {
273  value = EM_FRE;
274  } else if (lstr == "correspondence distance" || lstr == "correspondencedistance") {
275  value = EM_CorrespondenceDistance;
276  }
277  }
278 
279  // ---------------------------------------------------------------------------
280  // Alternative names for external point set forces
281  if (value == EM_Unknown) {
282  if (lstr == "edge force" || lstr == "edgeforce") value = EM_ImageEdgeForce;
283  }
284 
285  // ---------------------------------------------------------------------------
286  // Alternative names for internal point set forces
287  if (value == EM_Unknown) {
288  if (strcmp(str, "EdgeLength") == 0) value = EM_Stretching;
289  else if (strcmp(str, "MetricDistortion") == 0) value = EM_MetricDistortion;
290  else if (strcmp(str, "Bending") == 0) value = EM_Curvature;
291  else if (strcmp(str, "SurfaceBending") == 0) value = EM_Curvature;
292  else if (strcmp(str, "SurfaceCurvature") == 0) value = EM_Curvature;
293  else if (strcmp(str, "RepulsiveForce") == 0) value = EM_RepulsiveForce;
294  else if (strcmp(str, "NonSelfIntersection") == 0) value = EM_NonSelfIntersection;
295  else if (strcmp(str, "InflationForce") == 0) value = EM_InflationForce;
296  else if (strcmp(str, "SurfaceInflation") == 0) value = EM_InflationForce;
297  }
298 
299  // ---------------------------------------------------------------------------
300  // Alternative names for transformation regularization terms
301  if (value == EM_Unknown) {
302  if (lstr == "jac") value = EM_SqLogDetJac;
303  else if (lstr == "minjac") value = EM_MinDetJac;
304  }
305 
306  // ---------------------------------------------------------------------------
307  // Convert default names of energy measures
308  // (cf. ToString(EnergyMeasure) and ToPrettyString(EnergyMeasure))
309  if (value == EM_Unknown) {
310  string pretty;
311  value = static_cast<EnergyMeasure>(EM_Last - 1);
312  while (value != EM_Unknown) {
313  if (lstr == ToLower(ToString(value))) break;
314  pretty = ToLower(ToPrettyString(value));
315  if (lstr == pretty || lstr == TrimAll(pretty, " -")) break;
316  value = static_cast<EnergyMeasure>(value - 1);
317  }
318  }
319 
320  return (value != EM_Unknown);
321 }
