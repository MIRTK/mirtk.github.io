226 // -----------------------------------------------------------------------------
227 inline void LinearFreeFormTransformation4D
228 ::LocalTransform(double &x, double &y, double &z, double t, double) const
229 {
230  // Convert to lattice coordinates
231  double dx = x, dy = y, dz = z;
232  this->WorldToLattice(dx, dy, dz);
233  t = this->TimeToLattice(t);
234  // Evaluate displacement
235  this->Evaluate(dx, dy, dz, t);
236  // Transform point
237  x += dx, y += dy, z += dz;
238 }

243 {
244  // Convert to lattice coordinates

246  this->WorldToLattice(dx, dy, dz);

259 // ---------------------------------------------------------------------------
260 inline void LinearFreeFormTransformation4D
261 ::LocalJacobian(Matrix &jac, double x, double y, double z, double t, double) const
262 {
263  // Convert to lattice coordinates
264  this->WorldToLattice(x, y, z);
265  t = this->TimeToLattice(t);
266  // Compute 1st order derivatives
267  this->EvaluateJacobian(jac, x, y, z, t);
268  // Convert derivatives to world coordinates
269  JacobianToWorld(jac);
270  // Add derivatives of "x" term in T(x) = x + this->Evaluate(x)
271  jac(0, 0) += 1.0;
272  jac(1, 1) += 1.0;
273  jac(2, 2) += 1.0;
274 }

276 // -----------------------------------------------------------------------------
277 inline void LinearFreeFormTransformation4D
278 ::JacobianDOFs(double jac[3], int ci, int cj, int ck, int cl, double x, double y, double z, double t) const
279 {
280  // Convert point to lattice coordinates
281  this->WorldToLattice(x, y, z);
282  t = this->TimeToLattice(t);
283  // Calculate tensor product of 1D linear interpolation kernels
284  const double dx = abs(ci - x);
285  const double dy = abs(cj - y);
286  const double dz = abs(ck - z);
287  const double dt = abs(cl - t);
288  if (dx < 1.0 && dy < 1.0 && dz < 1.0 && dt < 1.0) {
289  jac[0] = jac[1] = jac[2] = (1.0 - dx) * (1.0 - dy) * (1.0 - dz) * (1.0 - dt);
290  } else {
291  jac[0] = jac[1] = jac[3] = .0;
292  }
293 }
