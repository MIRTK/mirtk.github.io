33 /**
34  * Compute curvature at each point of a surface mesh
35  *
36  * This curvature computation is based on a robust estimation of the 3D curvature
37  * tensor field as implemented in Matlab by Gabriel Peyre:
38  *
39  * http://uk.mathworks.com/matlabcentral/fileexchange/5355-toolbox-graph/content/toolbox_graph/compute_curvature.m
40  *
41  * The algorithm is detailed in
42  * David Cohen-Steiner and Jean-Marie Morvan.
43  * Restricted Delaunay triangulations and normal cycle.
44  * In Proc. 19th Annual ACM Symposium on Computational Geometry,
45  * pages 237-246, 2003.
46  * and also in
47  * Pierre Alliez, David Cohen-Steiner, Olivier Devillers, Bruno LeÂŽvy, and Mathieu Desbrun.
48  * Anisotropic Polygonal Remeshing.
49  * ACM Transactions on Graphics, 2003.
50  * Note: SIGGRAPH '2003 Conference Proceedings
51  *
52  * Alternatively, the vtkCurvatures filter can be used to compute mean curvature,
53  * Gauss curvature, and minimum and maximum curvature. To enable the use of this
54  * VTK filter, call VtkCurvaturesOn(). This option is ignored when the curvature
55  * tensor or principle directions are requested. If multiple curvature types
56  * supported by vtkCurvatures are requested, the minimum and maximum curvatures
57  * are computed by this filter from the mean and Gauss curvature values obtained
58  * by vtkCurvatures, to avoid the duplicate computation of these curvature
59  * values when using the vtkCurvatures filter. This computation is identical to
60  * vtkCurvatures::GetMinimumCurvature and vtkCurvatures::GetMaximumCurvature.
61  *
62  * @attention vtkPolyData::BuildLinks has to be called before filter execution.
63  */
64 class PolyDataCurvature : public PolyDataFilter

73  /// Enumeration of output curvature measures, can be combined using bitwise OR
74  enum Type
75  {
76  Minimum = 1, ///< Minimum principle curvature, k_min
77  Maximum = 2, ///< Maximum principle curvature, k_max
78  Mean = 4, ///< Mean curvature, (k_min + k_max)/2
79  Gauss = 8, ///< Gauss curvature, k_min * k_max
80  Curvedness = 16, ///< Curvedness, i.e., sqrt((k_min^2 + k_max^2) / 2)
81  Normal = 32, ///< Point normal computed from curvature tensor
82  MinimumDirection = 64, ///< Direction of minimum curvature
83  MaximumDirection = 128, ///< Direction of maximum curvature
84  Tensor = 256, ///< 6 entries of symmetric 3x3 curvature tensor
85  InverseTensor = 512, ///< 6 entries of inverse of symmetric 3x3 curvature tensor
86  Scalars = Minimum | Maximum | Mean | Gauss | Curvedness,
87  Directions = Normal | MinimumDirection | MaximumDirection
88  };
89 
90  /// Component indices of tensor output array
91  ///
92  /// This order is compatible with ParaView, which automatically labels the
93  /// entries in this order.
94  enum TensorIndex
95  {
96  XX = 0, YY = 1, ZZ = 2, XY = 3, YX = XY, YZ = 4, ZY = YZ, XZ = 5, ZX = XZ
97  };
