110 // -----------------------------------------------------------------------------
111 /// Convert connected components ordering to string
112 template <>
113 inline string ToString(const ConnectedComponentsOrdering &value, int w, char c, bool left)
114 {
115  const char *str;
116  switch (value) {
117  case CC_NoOrdering: str = "none"; break;
118  case CC_LargestFirst: str = "largest"; break;
119  case CC_SmallestFirst: str = "smallest"; break;
120  }
121  return ToString(str, w, c, left);
122 }
123 
124 // -----------------------------------------------------------------------------
125 /// Convert string to connected components ordering
126 template <>
127 inline bool FromString(const char *str, ConnectedComponentsOrdering &value)
128 {
129  const string lstr = ToLower(str);
130  if (lstr == "none" || lstr == "no ordering") {
131  value = CC_NoOrdering;
132  } else if (lstr == "largest" || lstr == "largestfirst" || lstr == "largest first") {
133  value = CC_LargestFirst;
134  } else if (lstr == "smallest" || lstr == "smallestfirst" || lstr == "smallest first") {
135  value = CC_SmallestFirst;
136  } else return false;
137  return true;
138 }
