86 template <class TImage>
87 inline void GenericInterpolateImageFunction<TImage>::Initialize(bool coeff)
88 {
89  // Ensure that input has the right type
90  if (!this->_Input) {
91  cerr << this->NameOfClass() << "::Initialize: Missing input image" << endl;
92  exit(1);
93  } else if (!dynamic_cast<const TImage *>(this->_Input)) {
94  cerr << this->NameOfClass() << "::Initialize: Invalid input image type" << endl;
95  exit(1);
96  }
97  // Initialize extrapolator, i.e., infinite discrete image
98  if (this->_InfiniteInput) {
99  if (!dynamic_cast<const ExtrapolatorType *>(this->_InfiniteInput)) {
100  cerr << this->NameOfClass() << "::Initialize: Invalid extrapolator type" << endl;
101  exit(1);
102  }
103  }
104  // Initialize base class
105  InterpolateImageFunction::Initialize(coeff);
106 }

195 template <class TImage>
196 GenericInterpolateImageFunction<TImage> *
197 GenericInterpolateImageFunction<TImage>
198 ::New(enum InterpolationMode mode, const TImage *image)
199 {
200  mode = InterpolationWithoutPadding(mode);
201 
202  GenericInterpolateImageFunction<TImage> *p = NULL;
203 
204  int dim = 0;
205  if (image) {
206  if (image->Z() == 1) dim = 2;
207  else if (image->T() == 1 || image->GetTSize() == .0) dim = 3;
208  else dim = 4;
209  }
210 
211  switch (dim) {
212  case 2: {
213  switch (mode) {
214  case Interpolation_NN:
215  p = new GenericNearestNeighborInterpolateImageFunction<TImage>();
216  break;
217  case Interpolation_Linear:
218  case Interpolation_FastLinear:
219  p = new GenericLinearInterpolateImageFunction2D<TImage>();
220  break;
221  case Interpolation_BSpline:
222  p = new GenericBSplineInterpolateImageFunction2D<TImage>();
223  break;
224  case Interpolation_CubicBSpline:
225  p = new GenericCubicBSplineInterpolateImageFunction2D<TImage>();
226  break;
227  case Interpolation_FastCubicBSpline:
228  p = new GenericFastCubicBSplineInterpolateImageFunction2D<TImage>();
229  break;
230  case Interpolation_CSpline:
231  p = new GenericCSplineInterpolateImageFunction2D<TImage>();
232  break;
233  case Interpolation_Gaussian:
234  p = new GenericGaussianInterpolateImageFunction2D<TImage>();
235  break;
236  case Interpolation_Sinc:
237  p = new GenericSincInterpolateImageFunction2D<TImage>();
238  break;
239  default:
240  p = NULL;
241  }
242  }
243  case 3: {
244  switch (mode) {
245  case Interpolation_NN:
246  p = new GenericNearestNeighborInterpolateImageFunction<TImage>();
247  break;
248  case Interpolation_Linear:
249  case Interpolation_FastLinear:
250  p = new GenericLinearInterpolateImageFunction3D<TImage>();
251  break;
252  case Interpolation_BSpline:
253  p = new GenericBSplineInterpolateImageFunction3D<TImage>();
254  break;
255  case Interpolation_CubicBSpline:
256  p = new GenericCubicBSplineInterpolateImageFunction3D<TImage>();
257  break;
258  case Interpolation_FastCubicBSpline:
259  p = new GenericFastCubicBSplineInterpolateImageFunction3D<TImage>();
260  break;
261  case Interpolation_CSpline:
262  p = new GenericCSplineInterpolateImageFunction3D<TImage>();
263  break;
264  case Interpolation_Gaussian:
265  p = new GenericGaussianInterpolateImageFunction3D<TImage>();
266  break;
267  case Interpolation_Sinc:
268  p = new GenericSincInterpolateImageFunction3D<TImage>();
269  break;
270  default:
271  p = NULL;
272  }
273  }
274  case 4: {
275  switch (mode) {
276  case Interpolation_NN:
277  p = new GenericNearestNeighborInterpolateImageFunction<TImage>();
278  break;
279  case Interpolation_Linear:
280  case Interpolation_FastLinear:
281  p = new GenericLinearInterpolateImageFunction4D<TImage>();
282  break;
283  case Interpolation_BSpline:
284  p = new GenericBSplineInterpolateImageFunction4D<TImage>();
285  break;
286  case Interpolation_CubicBSpline:
287  p = new GenericCubicBSplineInterpolateImageFunction4D<TImage>();
288  break;
289  case Interpolation_FastCubicBSpline:
290  p = new GenericFastCubicBSplineInterpolateImageFunction4D<TImage>();
291  break;
292  case Interpolation_CSpline:
293  p = new GenericCSplineInterpolateImageFunction4D<TImage>();
294  break;
295  case Interpolation_Gaussian:
296  p = new GenericGaussianInterpolateImageFunction4D<TImage>();
297  break;
298  case Interpolation_Sinc:
299  p = new GenericSincInterpolateImageFunction4D<TImage>();
300  break;
301  default:
302  p = NULL;
303  }
304  }
305  default: {
306  switch (mode) {
307  case Interpolation_NN:
308  p = new GenericNearestNeighborInterpolateImageFunction<TImage>();
309  break;
310  case Interpolation_Linear:
311  case Interpolation_FastLinear:
312  p = new GenericLinearInterpolateImageFunction<TImage>();
313  break;
314  case Interpolation_BSpline:
315  p = new GenericBSplineInterpolateImageFunction<TImage>();
316  break;
317  case Interpolation_CubicBSpline:
318  p = new GenericCubicBSplineInterpolateImageFunction<TImage>();
319  break;
320  case Interpolation_FastCubicBSpline:
321  p = new GenericFastCubicBSplineInterpolateImageFunction<TImage>();
322  break;
323  case Interpolation_CSpline:
324  p = new GenericCSplineInterpolateImageFunction<TImage>();
325  break;
326  case Interpolation_Gaussian:
327  p = new GenericGaussianInterpolateImageFunction<TImage>();
328  break;
329  case Interpolation_Sinc:
330  p = new GenericSincInterpolateImageFunction<TImage>();
331  break;
332  default:
333  p = NULL;
334  }
335  }
336  }
337 
338  // Initialize interpolator
339  if (p) {
340  p->NumberOfDimensions(dim);
341  p->Input(image);
342  // Throw error if no suitable interpolator available
343  } else {
344  cerr << "GenericInterpolateImageFunction::New: Interpolation mode (" << mode;
345  cerr << ") not supported for " << (dim ? ToString(dim) : "N") << "D images" << endl;
346  exit(1);
347  }
348 
349  return p;
350 }
