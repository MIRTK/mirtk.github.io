339 // -----------------------------------------------------------------------------
340 template <class HistogramType>
341 inline void Histogram1D<HistogramType>::AddSample(double x, HistogramType n)
342 {
343  if (x < _min || x > _max) return;
344  int index = iround(_nbins * (x - _min - 0.5*_width) / (_max - _min));
345  if (index < 0 ) index = 0;
346  if (index >= _nbins) index = _nbins - 1;
347  _bins[index] += n;
348  _nsamp += n;
349 }
350 
351 // -----------------------------------------------------------------------------
352 template <class HistogramType>
353 inline void Histogram1D<HistogramType>::DelSample(double x, HistogramType n)
354 {
355  if (x < _min || x > _max) return;
356  int index = iround(_nbins * (x - _min - 0.5*_width) / (_max - _min));
357  if (index < 0 ) index = 0;
358  if (index >= _nbins) index = _nbins - 1;
359  _bins[index] -= n;
360  _nsamp -= n;
361 }

399 // -----------------------------------------------------------------------------
400 template <class HistogramType>
401 inline double Histogram1D<HistogramType>::BinToCDF(int i) const
402 {
403  if (_nsamp == .0) return .0;
404  double s = .0;
405  for (int j = 0; j <= i; ++j) {
406  s += _bins[j];
407  }
408  return s / _nsamp;
409 }

418 // -----------------------------------------------------------------------------
419 template <class HistogramType>
420 inline int Histogram1D<HistogramType>::CDFToBin(double p) const
421 {
422  if (p < .0 || p > .1) {
423  cerr << "Histogram1D<HistogramType>::CDFToBin: Must be between 0 and 1" << endl;
424  exit(1);
425  }
426  int i;
427  double sum = .0;
428  for (i = 0; i < _nbins; ++i) {
429  sum += static_cast<double>(_bins[i]);
430  if (sum / _nsamp >= p) return i;
431  }
432  return _nbins - 1;
433 }
