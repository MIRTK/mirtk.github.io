1177 // -----------------------------------------------------------------------------
1178 inline bool FreeFormTransformation::BoundingBox(const Image *image, int cp,
1179  int &i1, int &j1, int &k1,
1180  int &i2, int &j2, int &k2,
1181  double fraction) const
1182 {
1183  // Calculate bounding box in world coordinates parallel to world axes
1184  double x[2], y[2], z[2];
1185  this->BoundingBox(cp, x[0], y[0], z[0], x[1], y[1], z[1], fraction);
1186 
1187  // Map bounding box into image space and calculate minimal axes-alinged
1188  // bounding box parallel to image axes which need not coincide with world axes
1189  Point p;
1190  double x1, y1, z1, x2, y2, z2;
1191  x1 = y1 = z1 = + numeric_limits<double>::infinity();
1192  x2 = y2 = z2 = - numeric_limits<double>::infinity();
1193 
1194  for (int c = 0; c <= 1; ++c)
1195  for (int b = 0; b <= 1; ++b)
1196  for (int a = 0; a <= 1; ++a) {
1197  p = Point(x[a], y[b], z[c]);
1198  image->WorldToImage(p);
1199  if (p._x < x1) x1 = p._x;
1200  if (p._x > x2) x2 = p._x;
1201  if (p._y < y1) y1 = p._y;
1202  if (p._y > y2) y2 = p._y;
1203  if (p._z < z1) z1 = p._z;
1204  if (p._z > z2) z2 = p._z;
1205  }
1206 
1207  // Round to nearest voxel in image domain
1208  i1 = static_cast<int>(round(x1));
1209  i2 = static_cast<int>(round(x2));
1210  j1 = static_cast<int>(round(y1));
1211  j2 = static_cast<int>(round(y2));
1212  k1 = static_cast<int>(round(z1));
1213  k2 = static_cast<int>(round(z2));
1214 
1215  // When both indices are outside in opposite directions,
1216  // use the full range [0, N[. If they are both outside in
1217  // the same direction, the condition i1 <= i2 is false which
1218  // indicates that the bounding box is empty in this case
1219  i1 = (i1 < 0 ? 0 : (i1 >= image->X() ? image->X() : i1));
1220  i2 = (i2 < 0 ? -1 : (i2 >= image->X() ? image->X() - 1 : i2));
1221  j1 = (j1 < 0 ? 0 : (j1 >= image->Y() ? image->Y() : j1));
1222  j2 = (j2 < 0 ? -1 : (j2 >= image->Y() ? image->Y() - 1 : j2));
1223  k1 = (k1 < 0 ? 0 : (k1 >= image->Z() ? image->Z() : k1));
1224  k2 = (k2 < 0 ? -1 : (k2 >= image->Z() ? image->Z() - 1 : k2));
1225  return i1 <= i2 && j1 <= j2 && k1 <= k2;
1226 }

1228 // -----------------------------------------------------------------------------
1229 inline bool FreeFormTransformation::BoundingBox(const Image *image, int cp,
1230  int &i1, int &j1, int &k1, int &l1,
1231  int &i2, int &j2, int &k2, int &l2,
1232  double fraction) const
1233 {
1234  // Calculate spatial bounding box in image coordinates
1235  bool bbvalid = BoundingBox(image, cp, i1, j1, k1, i2, j2, k2, fraction);
1236 
1237  // Calculate temporal bounding box
1238  double t1, t2;
1239  this->BoundingBox(cp, t1, t2, fraction);
1240 
1241  // Convert to image coordinates
1242  t1 = image->TimeToImage(t1);
1243  t2 = image->TimeToImage(t2);
1244  if (t2 < t1) swap(t1, t2);
1245 
1246  // Round to nearest voxel in image domain
1247  l1 = static_cast<int>(round(t1));
1248  l2 = static_cast<int>(round(t2));
1249 
1250  // When both indices are outside in opposite directions,
1251  // use the full range [0, N[. If they are both outside in
1252  // the same direction, the condition l1 <= l2 is false which
1253  // indicates that the bounding box is empty in this case
1254  l1 = (l1 < 0 ? 0 : (l1 >= image->T() ? image->T() : l1));
1255  l2 = (l2 < 0 ? -1 : (l2 >= image->T() ? image->T() - 1 : l2));
1256  return bbvalid && l1 <= l2;
1257 }

1467 {

1469  // w.r.t the world coordinates which are needed for the chain rule below
1470  const double &dudx = _matW2L(0, 0);
