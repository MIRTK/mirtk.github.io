35 /**
36  * Temporal diffeomorphic free-form transformation.
37  *
38  * This class implements a free-form transformation which is represented
39  * by a time-varying velocity field (3D+t). The 3D displacement field at a
40  * specific time is obtained by integrating the velocity field starting at the
41  * reference time point. The integration steps are adjusted if necessary in
42  * order to ensure that the resulting spatial transformation is diffeomorphic.
43  *
44  * For more details about the implementation see De Craene et al. (2012).
45  * Temporal diffeomorphic free-form deformation: application to motion and
46  * strain estimation from 3D echocardiography.
47  * Medical image analysis, 16(2), 427, 2012. doi:10.1016/j.media.2011.10.006
48  */

164 inline void LinearFreeFormTransformationTD::LocalTransform(double &x, double &y, double &z, double t1, double t2) const
165 {
166  double u, v, w, s;
167  Matrix sjac(3, 3);
168 
169  // Number of integration steps
170  int minsteps = abs(iround((t2 - t1) / _MaxTimeStep));
171  int maxsteps = abs(iround((t2 - t1) / _MinTimeStep));
172  if (minsteps < 2) minsteps = 2;
173  if (maxsteps < 2) maxsteps = 2;
174 
175  // Initial step size
176  int N = minsteps;
177  double dt = (t2 - t1) / N;
178  // Transform point by integrating the velocity over time
179  double t = t1;
180  for (int n = 0; n < N; n++) {
181  u = x;
182  v = y;
183  w = z;
184  this->WorldToLattice(u, v, w);
185  s = this->TimeToLattice(t);
186  // Check if still inside FFD domain
187  if (-2 <= u && u <= _x + 1 &&
188  -2 <= v && v <= _y + 1 &&
189  -2 <= w && w <= _z + 1 &&
190  -2 <= s && s <= _t + 1) {
191  // Reduce step size if necessary to ensure invertibility
192  while (N < maxsteps) {
193  // Compute current factor of Jacobian of deformation w.r.t the spatial variable.
194  //
195  // See equation (5) of De Craene et al., Medical Image Analysis 16 (2012).
196  // The partial derivatives are calculated w.r.t the spatial variables.
197  // The goal is to avoid foldings and singularities in the spatial domain
198  // while computing the temporal trajectory of a point.
199  EvaluateJacobian(sjac, u, v, w, s);
200  sjac = sjac * _matW2L(0, 0, 3, 3) * dt;
201  sjac(0, 0) += 1;
202  sjac(1, 1) += 1;
203  sjac(2, 2) += 1;
204  // In case of negative Jacobian determinant, reduce step size
205  if (sjac.Det3x3() < 0) {
206  N *= 2;
207  if (N > maxsteps) {
208  N = maxsteps;
209  dt = (t2 - t1) / N;
210  } else {
211  dt /= 2;
212  }
213  continue;
214  }
215  break;
216  }
217  // Integration step
218  EvaluateInside(u, v, w, s); // velocity
219  x += u * dt;
220  y += v * dt;
221  z += w * dt;
222  t += dt;
223  // Outside FFD domain, velocity is zero
224  } else {
225  break;
226  }
227  }
228 }
