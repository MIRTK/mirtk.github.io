146 // -----------------------------------------------------------------------------
147 template <>
148 inline string ToString(const LineSearchStrategy &m, int w, char c, bool left)
149 {
150  const char *str;
151  switch (m) {
152  case LS_None: str = "None"; break;
153  case LS_Adaptive: str = "Adaptive"; break;
154  case LS_Brent: str = "Brent"; break;
155  default: str = "Unknown"; break;
156  }
157  return ToString(str, w, c, left);
158 }
159 
160 // -----------------------------------------------------------------------------
161 template <>
162 inline bool FromString(const char *str, LineSearchStrategy &m)
163 {
164  m = LS_None;
165  if (strcmp(str, "None") == 0) return true;
166  if (strcmp(str, "linmin") == 0 || strcmp(str, "LinMin") == 0) m = LS_Brent;
167  if (m == LS_None) {
168  m = static_cast<LineSearchStrategy>(LS_Last - 1);
169  while (m != LS_None) {
170  if (ToString(m) == str) break;
171  m = static_cast<LineSearchStrategy>(m - 1);
172  }
173  }
174  return m != LS_None;
175 }
