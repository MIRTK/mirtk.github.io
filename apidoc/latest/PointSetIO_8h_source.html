300 /// Read polygonal dataset from GIFTI (.gii) file
301 ///
302 /// Standard GIFTI meta data is stored in the vtkInformation of the returned
303 /// vtkPolyData instance and the respective information of the corresponding
304 /// vtkDataArray instances of its point data.
305 ///
306 /// @param[in] fname File name.
307 /// @param[in,out] surface Polygonal dataset to which to add GIFTI data arrays.
308 /// If nullptr, a new polygonal dataset is allocated
309 /// and a GIFTI data array with @c NIFTI_INTENT_POINTSET
310 /// must be present in the GIFTI file. If input surface
311 /// has a previously read point set, the GIFTI file from
312 /// which to read additional point data can use sparse
313 /// storage using a data array with @c NIFTI_INTENT_NODE_INDEX
314 /// that specifies the points to which the read data array
315 /// values are added. Missing data values default to zero.
316 /// @param[in] errmsg Whether to print error messages if any.
317 ///
318 /// @return Polygonal dataset. Dataset is empty if file could not be read.
319 ///
320 /// Example:
321 /// @code
322 /// // Read geometry and topology of cortical surface model
323 /// vtkSmartPointer<vtkPolyData> surface = ReadGIFTI("cortex.surf.gii");
324 /// // Add shape measurements and functional statistics as surface point data
325 /// surface = ReadGIFTI("cortex.shape.gii", surface);
326 /// surface = ReadGIFTI("cortex.func.gii", surface);
327 /// @endcode
328 ///
329 /// @sa https://www.nitrc.org/projects/gifti/
330 vtkSmartPointer<vtkPolyData> ReadGIFTI(const char *fname,
331  vtkPolyData *surface = nullptr,
332  bool errmsg = false);
