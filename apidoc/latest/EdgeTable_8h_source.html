34 /**
35  * Edge table / adjacency matrix
36  *
37  * This class represents the adjacency matrix of point set nodes. It provides
38  * efficient access to the set of nodes adjacent to a given point. The non-zero
39  * entries stored in the sparse matrix are the one-based edge IDs such that
40  * sparse matrix entries are non-zero. To efficiently iterate all edges or a
41  * subset of these, use the thread-safe EdgeIterator.
42  */
43 class EdgeTable : public GenericSparseMatrix<int>
44 {

119 // -----------------------------------------------------------------------------
120 template <class EdgeIdType, class IdType1, class IdType2>
121 inline bool EdgeTable::GetEdge(EdgeIdType edgeId, IdType1 &ptId1, IdType2 &ptId2) const
122 {
123  const int eid = static_cast<int>(edgeId) + 1;
124  if (eid <= 0) return false;
125 
126  int i, j;
127  const int *row = _Row;
128  const int *col = _Col;
129  if (_Layout == CRS) swap(row, col);
130 
131  const IdType2 npoints = static_cast<IdType2>(NumberOfPoints());
132  for (ptId2 = 0; ptId2 < npoints; ++ptId2) {
133  for (i = col[ptId2], j = col[ptId2 + 1]; i < j; ++i) {
134  ptId1 = static_cast<IdType1>(row[i]);
135  if (_Data[i] == eid) return true;
136  if (ptId1 > ptId2) break;
137  }
138  }
139 
140  ptId1 = static_cast<IdType1>(-1);
141  ptId2 = static_cast<IdType2>(-1);
142  return false;
143 }

184 /**
185  * Thread-safe helper class for iteration of edges
186  */
187 class EdgeIterator
188 {
189  const EdgeTable &_Table;
190  int _EdgeId;
191  int _EndId;
192  const int *_PointId1;
193  const int *_ListEnd;
194  int _PointId2;
195 
196 public:
197 
198  /// Constructor
199  EdgeIterator(const EdgeTable &table)
200  :
201  _Table(table), _EdgeId(-1), _EndId(-1),
202  _PointId1(NULL), _ListEnd(NULL), _PointId2(-1)
203  {}
204 
205  /// Initialize traversal of edges
206  ///
207  /// @param[in] begin ID of first edge to iterate.
208  /// @param[in] end ID one behind last edge to iterate. If negative,
209  /// all edges from @p begin until the end are iterated.
210  ///
211  /// @bug Not sure if this functions works correctly for begin > 0, cf.,
212  /// AverageEdgeLength implementation.
213  inline void InitTraversal(int begin = 0, int end = -1)
214  {
215  _EdgeId = begin;
216  _EndId = ((end < 0 || end > _Table.NumberOfEdges()) ? _Table.NumberOfEdges() : end);
217  if (_EndId > _EdgeId) {
218  for (_PointId2 = 0; _PointId2 < _Table.NumberOfPoints(); ++_PointId2) {
219  _Table.GetAdjacentPoints(_PointId2, _PointId1, _ListEnd);
220  while (_PointId1 != _ListEnd) {
221  if (*_PointId1 > _PointId2) {
222  _PointId1 = _ListEnd;
223  break;
224  }
225  if (begin == 0) break;
226  ++_PointId1, --begin;
227  }
228  if (begin == 0 && _PointId1 != _ListEnd) break;
229  }
230  } else {
231  _PointId1 = _ListEnd = NULL;
232  _PointId2 = -1;
233  }
234  }
235 
236  /// Initialize traversal of edges in range
237  template <class IdType>
238  inline void InitTraversal(const blocked_range<IdType> &re)
239  {
240  InitTraversal(static_cast<int>(re.begin()), static_cast<int>(re.end()));
241  }
242 
243  /// Get next edge
244  ///
245  /// @param[out] ptId1 ID of first edge point.
246  /// @param[out] ptId2 ID of second edge point.
247  ///
248  /// @return ID of undirected edge (ptId1, ptId2) or -1 if end reached
249  template <class IdType>
250  int GetNextEdge(IdType &ptId1, IdType &ptId2)
251  {
252  if (_EdgeId >= _EndId) return -1;
253  ptId1 = static_cast<IdType>(*_PointId1);
254  ptId2 = static_cast<IdType>(_PointId2);
255  int edgeId = _EdgeId;
256  if (++_EdgeId < _EndId) {
257  if (++_PointId1 == _ListEnd || (*_PointId1) > _PointId2) {
258  do {
259  _Table.GetAdjacentPoints(++_PointId2, _PointId1, _ListEnd);
260  } while (_PointId1 == _ListEnd || (*_PointId1) > _PointId2);
261  }
262  }
263  return edgeId;
264  }
265 };
