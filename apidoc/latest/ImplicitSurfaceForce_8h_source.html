186 // -----------------------------------------------------------------------------
187 /// Convert implicit surface distance measure enumeration value to string
188 template <>
189 inline string ToString(const enum ImplicitSurfaceForce::DistanceMeasureType &value,
190  int w, char c, bool left)
191 {
192  const char *str;
193  switch (value) {
194  case ImplicitSurfaceForce::DM_Minimum: str = "Minimum"; break;
195  case ImplicitSurfaceForce::DM_Normal: str = "Normal"; break;
196  default: str = "Unknown"; break;
197  }
198  return ToString(str, w, c, left);
199 }
200 
201 // -----------------------------------------------------------------------------
202 /// Convert string to implicit surface distance measure enumeration value
203 template <>
204 inline bool FromString(const char *str, enum ImplicitSurfaceForce::DistanceMeasureType &value)
205 {
206  string lstr = ToLower(str);
207  if (lstr == "minimum") value = ImplicitSurfaceForce::DM_Minimum;
208  else if (lstr == "normal") value = ImplicitSurfaceForce::DM_Normal;
209  else value = ImplicitSurfaceForce::DM_Unknown;
210  return (value != ImplicitSurfaceForce::DM_Unknown);
211 }
