46 /**
47  * Point search structure for establishing point correspondences
48  *
49  * This point locator implementation is specialized for use by
50  * PointCorrespondence subclass implementations. It allows the search of
51  * nearest neighbors within the n-dimensional feature space spanned by the
52  * feature Arrays used to establish point correspondences.
53  *
54  * The implementation uses either VTK for point search in three dimensions or
55  * FLANN for higher dimensional feature spaces if available. Alternatively,
56  * an ITK Kd tree is used if the library is available. As last resort, a brute
57  * force search without actual Kd tree search structure is performed.
58  *
59  * \todo Consider actual implementation of own thread-safe N-dimensional Kd tree?
60  */
61 class PointLocator : public Object
62 {
63  mirtkObjectMacro(PointLocator);

660 // -----------------------------------------------------------------------------
661 inline int PointLocator::GetPointDimension(vtkPointSet *dataset, const FeatureList *features)
662 {
663  int dim;
664  if (features && features->size() > 0) {
665  dim = 0;
666  vtkDataArray *feature_array;
667  for (FeatureList::const_iterator feature = features->begin(); feature != features->end(); ++feature) {
668  if (feature->_Weight != .0) {
669  if (feature->_Index == -1) {
670  dim += 3;
671  } else {
672  feature_array = GetDataArray(dataset, *feature);
673  dim += feature_array->GetNumberOfComponents();
674  }
675  }
676  }
677  } else {
678  dim = 3;
679  }
680  return dim;
681 }

690 inline void PointLocator
691 ::GetPoint(Point &point, vtkPointSet *dataset, const Array<int> *sample, int index)
692 {
693  double pt[3];
694  const int i = GetPointIndex(dataset, sample, index);
695  dataset->GetPoint(i, pt);
696  point._x = pt[0], point._y = pt[1], point._z = pt[2];

699 // -----------------------------------------------------------------------------
700 inline void PointLocator
701 ::GetPoint(double *point, vtkPointSet *dataset, const Array<int> *sample, int index, const FeatureList *features)
702 {
703  const int i = GetPointIndex(dataset, sample, index);
704  if (features && features->size() > 0) {
705  vtkDataArray *feature_array;
706  for (FeatureList::const_iterator feature = features->begin(); feature != features->end(); ++feature) {
707  if (feature->_Weight != .0) {
708  if (feature->_Index == -1) {
709  dataset->GetPoint(i, point);
710  for (int d = 0; d < 3; ++d, ++point) {
711  (*point) = feature->_Weight * (feature->_Slope * (*point) + feature->_Intercept);
712  }
713  } else {
714  feature_array = GetDataArray(dataset, *feature);
715  feature_array->GetTuple(i, point);
716  for (int d = 0; d < feature_array->GetNumberOfComponents(); ++d, ++point) {
717  (*point) = feature->_Weight * (feature->_Slope * (*point) + feature->_Intercept);
718  }
719  }
720  }
721  }
722  } else {
723  dataset->GetPoint(i, point);
724  }
725 }

734 // -----------------------------------------------------------------------------
735 inline double PointLocator
736 ::Distance2BetweenPoints(const double *a, const double *b, int d)
737 {
738  double dx, dist2 = .0;
739  for (int i = 0; i < d; ++i) {
740  dx = b[i] - a[i];
741  dist2 += dx * dx;
742  }
743  return dist2;
744 }
