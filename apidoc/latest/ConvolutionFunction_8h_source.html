40 /// Mirror image at boundary
41 struct MirrorBoundaryCondition
42 {
43  /// Apply mirror boundary condition to voxel index
44  ///
45  /// \param[in] i Voxel index along image dimension
46  /// \param[in] N Number of voxels in image dimension
47  int operator ()(int i, int N) const
48  {
49  N -= 1;
50  int m, n;
51  if (i < 0) {
52  i = -i;
53  n = i / N;
54  m = i - n * N;
55  if (n % 2) i = N - 1 - m;
56  else i = m;
57  } else if (i > N) {
58  i = i - N;
59  n = i / N;
60  m = i - n * N;
61  if (n % 2) i = m;
62  else i = N - m;
63  }
64  return i;
65  }
66 
67  /// Apply mirror boundary condition to voxel pointer
68  template <class T>
69  const T *operator ()(int i, int N, const T *p, int stride)
70  {
71  return p + (this->operator ()(i, N) - i) * stride;
72  }
73 };

80 /// Perform convolution along x with truncation of kernel at boundary
81 template <class TKernel = double>
82 struct ConvolveInX : public VoxelFunction
83 {
84  const TKernel *_Kernel; ///< Convolution kernel
85  int _Radius; ///< Radius of kernel
86  bool _Normalize; ///< Wether to divide by sum of kernel weights
87  int _Offset; ///< Vector component offset
88  int _X; ///< Number of voxels in x
89 
90  ConvolveInX(const BaseImage *image, const TKernel *kernel, int size, bool norm = true, int l = 0)
91  :
92  _Kernel(kernel), _Radius((size - 1) / 2), _Normalize(norm),
93  _Offset(l * image->NumberOfSpatialVoxels()), _X(image->X())
94  {}
95 
96  template <class T1, class T2>
97  void operator ()(int i, int, int, int, const T1 *in, T2 *out) const
98  {
99  // Convolve l-th vector component
100  in += _Offset, out += _Offset;
101  // Go to start of input and end of kernel
102  i -= _Radius;
103  in -= _Radius;
104  int n = _Radius + _Radius/* + 1 - 1*/;
105  // Outside left boundary
106  if (i < 0) n += i, in -= i, i = 0;
107  // Inside image domain
108  double acc = .0, sum = .0;
109  while (i < _X && n >= 0) {
110  acc += static_cast<double>(_Kernel[n]) * static_cast<double>(*in);
111  sum += static_cast<double>(_Kernel[n]);
112  ++i, ++in, --n;
113  }
114  // Output result of convolution
115  if (_Normalize && sum != .0) acc /= sum;
116  *out = static_cast<T2>(acc);
117  }
118 };

430 struct ConvolveWeightedImageInX : public VoxelFunction
431 {
432  const TKernel *_Kernel; ///< Convolution kernel
433  int _Radius; ///< Radius of kernel
434  int _Offset1; ///< Image component offset
435  int _Offset2; ///< Weight component offset
436  int _X; ///< Number of voxels in x
437 
438  ConvolveWeightedImageInX(const BaseImage *image, const TKernel *kernel,
439  int size, int l1 = 0, int l2 = 0)
440  :

444  _X(image->X())
445  {}

447  template <class T1, class T2, class T3>
448  void operator ()(int i, int, int, int, const T1 *in, const T2 *win, T3 *out) const
449  {
450  // Convolve l1-th vector component using l2-th weight
451  in += _Offset1, win += _Offset2, out += _Offset1;
452  // Go to start of input and end of kernel
453  i -= _Radius;
454  in -= _Radius;
455  win -= _Radius;
456  int n = _Radius + _Radius/* + 1 - 1*/;
457  // Outside left boundary
458  if (i < 0) n += i, in -= i, win -= i, i = 0;
459  // Inside image domain
460  double w, acc = .0, sum = .0;
461  while (i < _X && n >= 0) {
462  w = static_cast<double>(*win) * static_cast<double>(_Kernel[n]);
463  acc += w * static_cast<double>(*in);
464  sum += w;
465  ++i, ++in, ++win, --n;
466  }
467  // Output result of convolution
468  acc = (sum > .0 ? acc / sum : .0);
469  *out = static_cast<T3>(acc);
470  }

609 // -----------------------------------------------------------------------------
610 /// Base class of 1D convolution functions which truncate the kernel at the foreground boundary
611 template <class TKernel = double>
612 class TruncatedForegroundConvolution1D : public VoxelFunction
613 {
614 protected:
615 
616  // ---------------------------------------------------------------------------
617  /// Constructor
618  TruncatedForegroundConvolution1D(const BaseImage *image, const TKernel *kernel, int size, bool norm = true)
619  :
620  _Kernel(kernel), _Size(size), _Radius((size - 1) / 2), _Normalize(norm),
621  _Background(image->GetBackgroundValueAsDouble())
622  {}
623 
624  // ---------------------------------------------------------------------------
625  /// Apply kernel initially at center voxel
626  template <class T>
627  bool ConvolveCenterVoxel(const T *in, double &acc, double &sum) const
628  {
629  if (*in == _Background) {
630  acc = _Background;
631  sum = .0;
632  return false;
633  } else {
634  acc = static_cast<double>(_Kernel[_Radius]) * static_cast<double>(*in);
635  sum = static_cast<double>(_Kernel[_Radius]);
636  return true;
637  }
638  }
639 
640  // ---------------------------------------------------------------------------
641  /// Apply kernel to left neighbors of given voxel
642  template <class T>
643  void ConvolveLeftNeighbors(int i, int n, const T *in, int s, double &acc, double &sum) const
644  {
645  int di = -1;
646  for (int k = _Radius + 1; k < _Size; ++k) {
647  // Move voxel index/pointer
648  i += di;
649  in -= s;
650  // Stop if outside image/foreground
651  if (i < 0 || i >= n || *in == _Background) break;
652  // Multiply with kernel weight
653  acc += static_cast<double>(_Kernel[k]) * static_cast<double>(*in);
654  sum += static_cast<double>(_Kernel[k]);
655  }
656  }
657 
658  // ---------------------------------------------------------------------------
659  /// Apply kernel to right neighbors of given voxel
660  template <class T>
661  void ConvolveRightNeighbors(int i, int n, const T *in, int s, double &acc, double &sum) const
662  {
663  int di = +1;
664  for (int k = _Radius - 1; k >= 0; --k) {
665  // Move voxel index/pointer
666  i += di;
667  in += s;
668  // Stop if outside image/foreground
669  if (i < 0 || i >= n || *in == _Background) break;
670  // Multiply with kernel weight
671  acc += static_cast<double>(_Kernel[k]) * static_cast<double>(*in);
672  sum += static_cast<double>(_Kernel[k]);
673  }
674  }
675 
676  // ---------------------------------------------------------------------------
677  template <class T>
678  void Put(T *out, double acc, double sum) const
679  {
680  if (_Normalize && sum != .0) acc /= sum;
681  (*out) = static_cast<T>(acc);
682  }
683 
684  // ---------------------------------------------------------------------------
685  // Data members
686 protected:
687 
688  const TKernel *_Kernel; ///< Convolution kernel
689  int _Size; ///< Size of kernel = 2 * _Radius + 1
690  int _Radius; ///< Radius of kernel
691  bool _Normalize; ///< Whether to normalize by sum of overlapping kernel weights
692  double _Background; ///< Background value (padding)
693 };
