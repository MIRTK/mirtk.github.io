33 /**
34  * Base class for image similarity measures
35  *
36  * The lower the value of the similarity measure, the more similar the images are.
37  * It may therefore more precisely be referred to as dissimilarity measure, but
38  * both terms are for historic reasons used interchangeably in this framework.
39  *
40  * If the transformed image has a foreground region defined, we assume that this
41  * corresponds to the region which needs to be matched with the untransformed
42  * image. Therefore we make use of the entire transformed foreground region to
43  * evaluate similarity and in particular the gradient forces defined within
44  * this region. Note that outside this source foreground region, the gradient
45  * is always zero and thus cannot be used to drive the image registration.
46  *
47  * Otherwise, the region of interest is generally defined by the foreground of
48  * the untransformed image. For each such target voxel we want to find a suitable
49  * corresponding voxel in the other image. Therefore, we restrict the similarity
50  * evaluation to this region. In case of a symmetric transformation of both
51  * images, the union of the foreground of both images defines the region for
52  * which similarity is evaluated and forces are computed.
53  *
54  * Subclasses which implement a particular similarity measure should call
55  * the IsForeground member function to decide whether or not to consider
56  * a given voxel of the grid on which the registered images are defined.
57  *
58  * \note The similarity measure owns the registered input images and may thus
59  * modify these to improve the runtime of each Update step.
60  */

235  /// Multiply voxel-wise similarity gradient by transformed image gradient
236  ///
237  /// This function is intended for use by subclass implementations to compute
238  /// the NonParametericGradient. It applies the chain rule to compute
239  /// \f$\frac{dSimilarity}{dy} = \frac{dSimilarity}{dI} * \frac{dI}{dy}\f$, given
240  /// \f$\frac{dSimilarity}{dI}\f$ as input, where \f$y = T(x)\f$.
241  ///
242  /// \param[in] image Transformed image.
243  /// \param[in,out] gradient Input must be the gradient of the image similarity
244  /// w.r.t. the transformed \p image in x. Output is the
245  /// voxel-wise gradient of the similarity w.r.t. T(x).
246  void MultiplyByImageGradient(const RegisteredImage *image,
247  GradientImageType *gradient);

249  /// Compute voxel-wise non-parametric similarity gradient w.r.t the given image
250  ///
251  /// Must be implemented by subclasses to compute the similarity gradient.
252  /// The base class implementation can be used to convert the similarity gradient
253  /// computed w.r.t transformed image (i.e., dSimilarity/dI) to a voxel-wise
254  /// non-parametric gradient (i.e., dSimilarity/dT). Note that the input must
255  /// be a scalar field only. The base class copies the x component of the input

257  ///

339  /// Evaluate similarity gradient
340  ///
341  /// This function calls the virtual NonParametricGradient function to be
342  /// implemented by subclasses for each transformed input image to obtain
343  /// the voxel-wise similarity gradient. It then converts this gradient into
344  /// a gradient w.r.t the transformation parameters using the ParametricGradient.
345  ///
346  /// If both target and source are transformed by different transformations,
347  /// the resulting gradient vector contains first the derivative values w.r.t
348  /// the parameters of the target transformation followed by those computed
349  /// w.r.t the parameters of the source transformation. If both images are
350  /// transformed by the same transformation, the sum of the derivative values
351  /// is added to the resulting gradient vector. This is in particular the case
352  /// for a velocity based transformation model which is applied to deform both
353  /// images "mid-way". Otherwise, only one input image is transformed
354  /// (usually the source) and the derivative values of only the respective
355  /// transformation parameters added to the gradient vector.
356  ///
357  /// \sa NonParametricGradient, ParametricGradient
358  ///
359  /// \param[in] image Transformed image.
360  /// \param[in,out] np_gradient Memory for voxel-wise non-parametric gradient.
361  /// \param[in,out] gradient Gradient to which the computed gradient of the
362  /// image similarity is added after multiplying by
363  /// the given similarity \p weight.
364  /// \param[in] step Step size to use for finite differences.
365  /// \param[in] weight Weight of image similarity.
366  virtual void EvaluateGradient(RegisteredImage *image,
367  GradientImageType *&np_gradient,
368  double *gradient,
369  double step, double weight);

419 inline bool ImageSimilarity::IsForeground(int idx) const
420 {
421  // Never evaluate similarity outside explicitly specified domain
422  if (_Mask && !_Mask->Get(idx)) return false;
423  // If both images are transformed (symmetric registration)...
424  if (_Target->Transformation() && _Source->Transformation()) {
425  // ... evaluate within union of foreground regions
426  return _Target->IsForeground(idx) || _Source->IsForeground(idx);
427  // If second image is transformed...
428  } else if (_Source->Transformation()) {
429  // ... try to match masked region of transformed image if given
430  if (_Source->HasMask()) return _Source->IsForeground(idx);
431  // ... otherwise evaluate similarity for each untransformed voxel,
432  // but consider that similarity gradient is anyway zero for voxels
433  // that are mapped outside the transformed image foreground
434  return _Source->IsForeground(idx) && _Target->IsForeground(idx);
435  // If first image is transformed same as above with reversed order though
436  } else {
437  if (_Target->HasMask()) return _Target->IsForeground(idx);
438  return _Target->IsForeground(idx) && _Source->IsForeground(idx);
439  }
440 }

443 inline bool ImageSimilarity::IsForeground(int i, int j, int k) const
444 {
445  // Never evaluate similarity outside explicitly specified domain
446  if (_Mask && !_Mask->Get(i, j, k)) return false;
447  // If both images are transformed (symmetric registration)...
448  if (_Target->Transformation() && _Source->Transformation()) {
449  // ... evaluate within union of foreground regions
450  return _Target->IsForeground(i, j, k) || _Source->IsForeground(i, j, k);
451  // If second image is transformed...
452  } else if (_Source->Transformation()) {
453  // ... try to match masked region of transformed image if given
454  if (_Source->HasMask()) return _Source->IsForeground(i, j, k);
455  // ... otherwise evaluate similarity for each untransformed voxel,
456  // but consider that similarity gradient is anyway zero for voxels
457  // that are mapped outside the transformed image foreground
458  return _Source->IsForeground(i, j, k) && _Target->IsForeground(i, j, k);
459  // If first image is transformed same as above with reversed order though
460  } else {
461  if (_Target->HasMask()) return _Target->IsForeground(i, j, k);
462  return _Target->IsForeground(i, j, k) && _Source->IsForeground(i, j, k);
