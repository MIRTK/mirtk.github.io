138  /// Test stopping criteria
139  ///
140  /// The objective function must be up-to-date when this function is called.
141  /// This is usually the case anyway because the current objective function
142  /// value after a change of the parameters must be evaluated before this
143  /// function is called to be able to provide this value as argument.
144  ///
145  /// \note The objective function value may be NaN in case of a non-parametric
146  /// deformable surface model. In this case, stopping criteria are based
147  /// only on the current surface geometry or last node displacements.
148  /// Stopping criteria based on the objective function value should
149  /// never be fulfilled in this case and always return \c false.
150  ///
151  /// \param[in] iter Current number of iterations.
152  /// \param[in] prev Objective function value at previous iteration.
153  /// \param[in] value Objective function value at current iteration.
154  /// \param[in] delta Last change of objective function parameters.
155  ///
156  /// \returns True when at least one stopping criterion is fulfilled.
157  virtual bool Converged(int iter, double prev, double value, const double *delta);

203 // -----------------------------------------------------------------------------
204 inline bool LocalOptimizer::IsImprovement(double prev, double value) const
205 {
206  if (IsInf(prev)) return !IsNaN(value);
207  if (IsInf(value)) return false;
208  return prev - value > _Epsilon;
209 }
